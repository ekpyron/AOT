220AOT_axiom indistinguishable_ord_enc_all: ‹Π↓ & A!x & A!y & ∀F □([F]x ≡ [F]y) →
221  ((∀G(∀z (O!z → □([G]z ≡ [Π]z)) → x[G])) ≡ ∀G(∀z(O!z → □([G]z ≡ [Π]z)) → y[G]))›
222proof (rule AOT_model_axiomI)
223  AOT_modally_strict {
224    {
225      AOT_assume Π_den: ‹[Π]↓›
226      AOT_assume ‹A!x›
227      AOT_hence 0: ‹[λx ¬◇[«AOT_sem_concrete»]x]x›
228        by (simp add: AOT_concrete_sem AOT_sem_abstract)
229      AOT_assume ‹A!y›
230      AOT_hence 1: ‹[λx ¬◇[«AOT_sem_concrete»]x]y›
231        by (simp add: AOT_concrete_sem AOT_sem_abstract)
232      AOT_assume 2: ‹∀F □([F]x ≡ [F]y)›
233      {
234        AOT_assume ‹∀G(∀z (O!z → □([G]z ≡ [Π]z)) → x[G])›
235        AOT_hence 3: ‹∀G(∀z ([λx ◇[«AOT_sem_concrete»]x]z → □([G]z ≡ [Π]z)) → x[G])›
236          by (simp add: AOT_concrete_sem AOT_sem_ordinary)
237        {
238          fix Π' :: ‹<κ>›
239          AOT_assume 1: ‹Π'↓›
240          AOT_assume 2: ‹[λx ◇[«AOT_sem_concrete»]x]z → □([Π']z ≡ [Π]z)› for z
241          AOT_have ‹x[Π']›
242            using 3
243            by (auto simp: AOT_sem_forall AOT_sem_imp AOT_sem_box AOT_sem_denotes)
244               (metis (no_types, lifting) 1 2 AOT_term_of_var_cases AOT_sem_box AOT_sem_denotes AOT_sem_imp)
245        } note 3 = this
246        fix Π' :: ‹<κ>›
247        AOT_assume Π_den: ‹Π'↓›
248        AOT_assume 4: ‹∀z (O!z → □([Π']z ≡ [Π]z))›
249        {
250          fix κ⇩0
251          AOT_assume ‹[λx ◇[«AOT_sem_concrete»]x]κ⇩0›
252          AOT_hence ‹O!κ⇩0›
253            using AOT_concrete_sem AOT_sem_ordinary by simp
254          moreover AOT_have ‹κ⇩0↓›
255            using calculation by (simp add: AOT_sem_exe)
256          ultimately AOT_have ‹□([Π']κ⇩0 ≡ [Π]κ⇩0)› using 4 by (auto simp: AOT_sem_forall AOT_sem_imp)
257        } note 4 = this
258        AOT_have ‹y[Π']›
259          apply (rule AOT_sem_enc_indistinguishable_all)
260          apply (fact 0)
261               apply (auto simp: 0 1 Π_den 2[simplified AOT_sem_forall AOT_sem_box AOT_sem_equiv])
262          apply (rule 3)
263           apply auto[1]
264          using 4
265          by (auto simp: AOT_sem_imp AOT_sem_equiv AOT_sem_box)
266      }
267      AOT_hence ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → y[G])› if ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → x[G])›
268        using that
269        by (auto simp: AOT_sem_forall AOT_sem_imp)
270      moreover {
271      {
272        AOT_assume ‹∀G(∀z (O!z → □([G]z ≡ [Π]z)) → y[G])›
273        AOT_hence 3: ‹∀G(∀z ([λx ◇[«AOT_sem_concrete»]x]z → □([G]z ≡ [Π]z)) → y[G])›
274          by (simp add: AOT_concrete_sem AOT_sem_ordinary)
275        {
276          fix Π' :: ‹<κ>›
277          AOT_assume 1: ‹Π'↓›
278          AOT_assume 2: ‹[λx ◇[«AOT_sem_concrete»]x]z → □([Π']z ≡ [Π]z)› for z
279          AOT_have ‹y[Π']›
280            using 3 apply (auto simp: AOT_sem_forall AOT_sem_imp AOT_sem_box AOT_sem_denotes)
281            by (metis (no_types, lifting) 1 2 AOT_model.AOT_term_of_var_cases AOT_sem_box AOT_sem_denotes AOT_sem_imp)
282        } note 3 = this
