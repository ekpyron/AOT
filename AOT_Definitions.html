26  have "conventions3[1]": ‹«φ → ψ ≡ ¬ψ → ¬φ» = «(φ → ψ) ≡ (¬ψ → ¬φ)»›
27    by blast
28  have "conventions3[2]": ‹«φ & ψ → χ» = «(φ & ψ) → χ»›
29                   and ‹«φ ∨ ψ → χ» = «(φ ∨ ψ) → χ»›
30    by blast+
31  have "conventions3[3]": ‹«φ ∨ ψ & χ» = «(φ ∨ ψ) & χ»›
32                   and ‹«φ & ψ ∨ χ» = «(φ & ψ) ∨ χ»›
33     by blast+ ― ‹Note that PLM instead generally uses parenthesis in these cases.›
