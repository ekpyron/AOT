46AOT_theorem ConceptOfOrdinaryProperty: ‹([H] ⇒ O!) → [λx ConceptOf(x,H)]↓›
47proof(rule "→I")
48  AOT_assume ‹[H] ⇒ O!›
49  AOT_hence ‹□∀x([H]x → O!x)›
50    using "F-imp-G"[THEN "≡⇩d⇩fE"] "&E" by blast
51  AOT_hence ‹□□∀x([H]x → O!x)›
52    using "S5Basic:6"[THEN "≡E"(1)] by blast
53  moreover AOT_have ‹□□∀x([H]x → O!x) → □∀F∀G(□(G ≡⇩E F) → ([H] ⇒ [F] ≡ [H] ⇒ [G]))›
54  proof(rule RM; safe intro!: "→I" GEN "≡I")
55    AOT_modally_strict {
56      fix F G
57      AOT_assume 0: ‹□∀x([H]x → O!x)›
58      AOT_assume ‹□G ≡⇩E F›
59      AOT_hence 1: ‹□∀u([G]u ≡ [F]u)›
60        by (AOT_subst_thm eqE[THEN "≡Df", THEN "≡S"(1), OF "&I",
61              OF "cqt:2[const_var]"[axiom_inst],
62              OF "cqt:2[const_var]"[axiom_inst], symmetric])
63      {
64        AOT_assume ‹[H] ⇒ [F]›
65        AOT_hence ‹□∀x([H]x → [F]x)›
66          using "F-imp-G"[THEN "≡⇩d⇩fE"] "&E" by blast
67        moreover AOT_modally_strict {
68          AOT_assume ‹∀x([H]x → O!x)›
69          moreover AOT_assume ‹∀u([G]u ≡ [F]u)›
70          moreover AOT_assume ‹∀x([H]x → [F]x)›
71          ultimately AOT_have ‹[H]x → [G]x› for x
72            by (auto intro!: "→I" dest!: "∀E"(2) dest: "→E" "≡E")
73          AOT_hence ‹∀x([H]x → [G]x)›
74            by (rule GEN)
75        }
76        ultimately AOT_have ‹□∀x([H]x → [G]x)›
77          using "RN[prem]"[where
78              Γ="{«∀x([H]x → O!x)», «∀u([G]u ≡ [F]u)»,«∀x([H]x → [F]x)»}"]
79          using 0 1 by fast
80        AOT_thus ‹[H] ⇒ [G]›
81          by (AOT_subst_def "F-imp-G")
82             (safe intro!: "cqt:2" "&I")
83      }
84      {
85        AOT_assume ‹[H] ⇒ [G]›
86        AOT_hence ‹□∀x([H]x → [G]x)›
87          using "F-imp-G"[THEN "≡⇩d⇩fE"] "&E" by blast
88        moreover AOT_modally_strict {
89          AOT_assume ‹∀x([H]x → O!x)›
90          moreover AOT_assume ‹∀u([G]u ≡ [F]u)›
91          moreover AOT_assume ‹∀x([H]x → [G]x)›
92          ultimately AOT_have ‹[H]x → [F]x› for x
93            by (auto intro!: "→I" dest!: "∀E"(2) dest: "→E" "≡E")
94          AOT_hence ‹∀x([H]x → [F]x)›
95            by (rule GEN)
96        }
97        ultimately AOT_have ‹□∀x([H]x → [F]x)›
98          using "RN[prem]"[where
99              Γ="{«∀x([H]x → O!x)», «∀u([G]u ≡ [F]u)»,«∀x([H]x → [G]x)»}"]
100          using 0 1 by fast
101        AOT_thus ‹[H] ⇒ [F]›
102          by (AOT_subst_def "F-imp-G")
103             (safe intro!: "cqt:2" "&I")
104      }
105    }
106  qed
107  ultimately AOT_have ‹□∀F∀G(□(G ≡⇩E F) → ([H] ⇒ [F] ≡ [H] ⇒ [G]))›
108    using "→E" by blast
109  AOT_hence 0: ‹[λx ∀F(x[F] ≡ ([H] ⇒ [F]))]↓›
110    using Comprehension_3[THEN "→E"] by blast
111  AOT_show ‹[λx ConceptOf(x,H)]↓›
112  proof (rule "safe-ext"[axiom_inst, THEN "→E", OF "&I"])
113    AOT_show ‹[λx C!x & [λx ∀F(x[F] ≡ ([H] ⇒ [F]))]x]↓› by "cqt:2"
114  next
115    AOT_show ‹□∀x (C!x & [λx ∀F (x[F] ≡ [H] ⇒ [F])]x ≡ ConceptOf(x,H))›
116    proof (rule "RN[prem]"[where Γ=‹{«[λx ∀F(x[F] ≡ ([H] ⇒ [F]))]↓»}›, simplified])
117      AOT_modally_strict {
118        AOT_assume 0: ‹[λx ∀F (x[F] ≡ [H] ⇒ [F])]↓›
119        AOT_show ‹∀x (C!x & [λx ∀F (x[F] ≡ [H] ⇒ [F])]x ≡ ConceptOf(x,H))›
120        proof(safe intro!: GEN "≡I" "→I" "&I")
121          fix x
122          AOT_assume ‹C!x & [λx ∀F (x[F] ≡ [H] ⇒ [F])]x›
123          AOT_thus ‹ConceptOf(x,H)›
124            by (AOT_subst_def "concept-of-G")
125               (auto intro!: "&I" "cqt:2" dest: "&E" "β→C")
126        next
127          fix x
128          AOT_assume ‹ConceptOf(x,H)›
129          AOT_hence ‹C!x & (H↓ & ∀F(x[F] ≡ [H] ⇒ [F]))›
130            by (AOT_subst_def (reverse) "concept-of-G")
131          AOT_thus ‹C!x› and ‹[λx ∀F(x[F] ≡ [H] ⇒ [F])]x›
132            by (auto intro!: "β←C" 0 "cqt:2" dest: "&E")
133        qed
134      }
135    next
136      AOT_show ‹□[λx ∀F(x[F] ≡ ([H] ⇒ [F]))]↓›
137        using "exist-nec"[THEN "→E"] 0 by blast
138    qed
139  qed
140qed

142AOT_define ThickForm :: ‹τ ⇒ τ ⇒ φ› (‹FormOf'(_,_')›)
143  "tform-of": ‹FormOf(x,G) ≡⇩d⇩f A!x & G↓ & ∀F(x[F] ≡ [G] ⇒ [F])›
144
145AOT_theorem FormOfOrdinaryProperty: ‹([H] ⇒ O!) → [λx FormOf(x,H)]↓›
146proof(rule "→I")
147  AOT_assume 0: ‹[H] ⇒ [O!]›
148  AOT_show ‹[λx FormOf(x,H)]↓›
149  proof (rule "safe-ext"[axiom_inst, THEN "→E", OF "&I"])
150    AOT_show ‹[λx ConceptOf(x,H)]↓›
151      using 0 ConceptOfOrdinaryProperty[THEN "→E"] by blast
152    AOT_show ‹□∀x (ConceptOf(x,H) ≡ FormOf(x,H))›
153    proof(safe intro!: RN GEN)
154      AOT_modally_strict {
155        fix x
156        AOT_modally_strict {
157          AOT_have ‹A!x ≡ A!x›
158            by (simp add: "oth-class-taut:3:a")
159          AOT_hence ‹C!x ≡ A!x›
160            using "rule-id-df:1[zero]"[OF concepts, OF "oa-exist:2"]
161                  "rule=E" id_sym by fast
162        }
163        AOT_thus ‹ConceptOf(x,H) ≡ FormOf(x,H)›
164          by (AOT_subst_def "tform-of";
165              AOT_subst_def "concept-of-G";
166              AOT_subst ‹C!x› ‹A!x›)
167             (auto intro!: "≡I" "→I" "&I" dest: "&E")
168      }
169    qed
170  qed
171qed
