(* conventions - these are already defined, resp. valid, so just note them here again *)
notepad
begin
  fix φ ψ χ

  AOT_have ‹φ & ψ ≡⇩d⇩f ¬(φ → ¬ψ)›
    using "conventions:1" .
  AOT_have ‹φ ∨ ψ ≡⇩d⇩f ¬φ → ψ›
    using "conventions:2" .
  AOT_have ‹φ ≡ ψ ≡⇩d⇩f (φ → ψ) & (ψ → φ)›
    using "conventions:3" .
  {
    fix φ :: ‹'a::AOT_Term ⇒ 𝗈›
    AOT_have ‹∃α φ{α} ≡⇩d⇩f ¬∀α ¬φ{α}›
      using "conventions:4" .
  }
  AOT_have ‹◇φ ≡⇩d⇩f ¬□¬φ›
    using "conventions:5" .

  have "conventions3[1]": ‹«φ → ψ ≡ ¬ψ → ¬φ» = «(φ → ψ) ≡ (¬ψ → ¬φ)»› by blast
  have "conventions3[2]": ‹«φ & ψ → χ» = «(φ & ψ) → χ»›
                   and ‹«φ ∨ ψ → χ» = «(φ ∨ ψ) → χ»›
    by blast+
  have "conventions3[3]": ‹«φ ∨ ψ & χ» = «(φ ∨ ψ) & χ»›
                   and ‹«φ & ψ ∨ χ» = «(φ & ψ) ∨ χ»› (* not exactly, but close enough *)
     by blast+
end

AOT_axiom indistinguishable_ord_enc_all: ‹Π↓ & A!x & A!y & ∀F □([F]x ≡ [F]y) →
  ((∀G(∀z (O!z → □([G]z ≡ [Π]z)) → x[G])) ≡ ∀G(∀z(O!z → □([G]z ≡ [Π]z)) → y[G]))›
proof (rule AOT_model_axiomI)
  AOT_modally_strict {
    {
      AOT_assume Π_den: ‹[Π]↓›
      AOT_assume ‹A!x›
      AOT_hence 0: ‹[λx ¬◇[«AOT_sem_concrete»]x]x›
        by (simp add: AOT_concrete_sem AOT_sem_abstract)
      AOT_assume ‹A!y›
      AOT_hence 1: ‹[λx ¬◇[«AOT_sem_concrete»]x]y›
        by (simp add: AOT_concrete_sem AOT_sem_abstract)
      AOT_assume 2: ‹∀F □([F]x ≡ [F]y)›
      {
        AOT_assume ‹∀G(∀z (O!z → □([G]z ≡ [Π]z)) → x[G])›
        AOT_hence 3: ‹∀G(∀z ([λx ◇[«AOT_sem_concrete»]x]z → □([G]z ≡ [Π]z)) → x[G])›
          by (simp add: AOT_concrete_sem AOT_sem_ordinary)
        {
          fix Π' :: ‹<κ>›
          AOT_assume 1: ‹Π'↓›
          AOT_assume 2: ‹[λx ◇[«AOT_sem_concrete»]x]z → □([Π']z ≡ [Π]z)› for z
          AOT_have ‹x[Π']›
            using 3 apply (auto simp: AOT_sem_forall AOT_sem_imp AOT_sem_box AOT_sem_denotes)
            by (metis (no_types, lifting) 1 2 AOT_model.AOT_term_of_var_cases AOT_sem_box AOT_sem_denotes AOT_sem_imp)
        } note 3 = this
        fix Π' :: ‹<κ>›
        AOT_assume Π_den: ‹Π'↓›
        AOT_assume 4: ‹∀z (O!z → □([Π']z ≡ [Π]z))›
        {
          fix κ⇩0
          AOT_assume ‹[λx ◇[«AOT_sem_concrete»]x]κ⇩0›
          AOT_hence ‹O!κ⇩0›
            using AOT_concrete_sem AOT_sem_ordinary by simp
          moreover AOT_hence ‹κ⇩0↓›
            by (simp add: AOT_sem_exe)
          ultimately AOT_have ‹□([Π']κ⇩0 ≡ [Π]κ⇩0)› using 4 by (auto simp: AOT_sem_forall AOT_sem_imp)
        } note 4 = this
        AOT_have ‹y[Π']›
          apply (rule AOT_sem_enc_indistinguishable_all)
          apply (fact 0)
               apply (auto simp: 0 1 Π_den 2[simplified AOT_sem_forall AOT_sem_box AOT_sem_equiv])
          apply (rule 3)
           apply auto[1]
          using 4
          by (auto simp: AOT_sem_imp AOT_sem_equiv AOT_sem_box)
      }
      AOT_hence ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → y[G])› if ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → x[G])›
        using that
        by (auto simp: AOT_sem_forall AOT_sem_imp)
      moreover {
      {
        AOT_assume ‹∀G(∀z (O!z → □([G]z ≡ [Π]z)) → y[G])›
        AOT_hence 3: ‹∀G(∀z ([λx ◇[«AOT_sem_concrete»]x]z → □([G]z ≡ [Π]z)) → y[G])›
          by (simp add: AOT_concrete_sem AOT_sem_ordinary)
        {
          fix Π' :: ‹<κ>›
          AOT_assume 1: ‹Π'↓›
          AOT_assume 2: ‹[λx ◇[«AOT_sem_concrete»]x]z → □([Π']z ≡ [Π]z)› for z
          AOT_have ‹y[Π']›
            using 3 apply (auto simp: AOT_sem_forall AOT_sem_imp AOT_sem_box AOT_sem_denotes)
            by (metis (no_types, lifting) 1 2 AOT_model.AOT_term_of_var_cases AOT_sem_box AOT_sem_denotes AOT_sem_imp)
        } note 3 = this
        fix Π' :: ‹<κ>›
        AOT_assume Π_den: ‹Π'↓›
        AOT_assume 4: ‹∀z (O!z → □([Π']z ≡ [Π]z))›
        {
          fix κ⇩0
          AOT_assume ‹[λx ◇[«AOT_sem_concrete»]x]κ⇩0›
          AOT_hence ‹O!κ⇩0›
            using AOT_concrete_sem AOT_sem_ordinary by simp
          moreover AOT_hence ‹κ⇩0↓›
            by (simp add: AOT_sem_exe)
          ultimately AOT_have ‹□([Π']κ⇩0 ≡ [Π]κ⇩0)› using 4 by (auto simp: AOT_sem_forall AOT_sem_imp)
        } note 4 = this
        AOT_have ‹x[Π']›
          apply (rule AOT_sem_enc_indistinguishable_all)
          apply (fact 1)
               apply (auto simp: 0 1 Π_den 2[simplified AOT_sem_forall AOT_sem_box AOT_sem_equiv])
          apply (rule 3)
           apply auto[1]
          using 4
          by (auto simp: AOT_sem_imp AOT_sem_equiv AOT_sem_box)
      }

      AOT_hence ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → x[G])› if ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → y[G])›
        using that
        by (auto simp: AOT_sem_forall AOT_sem_imp)
      }
      ultimately AOT_have ‹∀G (∀z (O!z → □([G]z ≡ [Π]z)) → x[G]) ≡ ∀G (∀z (O!z → □([G]z ≡ [Π]z)) → y[G])›
        by (auto simp: AOT_sem_equiv)
    }
    AOT_thus ‹Π↓ & A!x & A!y & ∀F □([F]x ≡ [F]y) →
       (∀G (∀z (O!z → □([G]z ≡ [Π]z)) → x[G]) ≡ ∀G (∀z (O!z → □([G]z ≡ [Π]z)) → y[G]))›
      by (auto simp: AOT_sem_imp AOT_sem_conj)
  }
qed

AOT_axiom indistinguishable_ord_enc_ex: ‹Π↓ & A!x & A!y & ∀F □([F]x ≡ [F]y) →
  ((∃G(∀z (O!z → □([G]z ≡ [Π]z)) & x[G])) ≡ ∃G(∀z(O!z → □([G]z ≡ [Π]z)) & y[G]))›
proof (rule AOT_model_axiomI)
  have Aux: ‹[v ⊨ [λx ◇[«AOT_sem_concrete»]x]κ] = ([v ⊨ [λx ◇[«AOT_sem_concrete»]x]κ] ∧ [v ⊨ κ↓])› for v κ
    using AOT_sem_exe by blast
  AOT_modally_strict {
    fix x y
    AOT_assume Π_den: ‹[Π]↓›
    AOT_assume 2: ‹∀F □([F]x ≡ [F]y)›
    AOT_assume ‹A!x›
    AOT_hence 0: ‹[λx ¬◇[«AOT_sem_concrete»]x]x›
      by (simp add: AOT_concrete_sem AOT_sem_abstract)
    AOT_assume ‹A!y›
    AOT_hence 1: ‹[λx ¬◇[«AOT_sem_concrete»]x]y›
      by (simp add: AOT_concrete_sem AOT_sem_abstract)
    {
      AOT_assume ‹∃G(∀z (O!z → □([G]z ≡ [Π]z)) & x[G])›
      then AOT_obtain Π' where Π'_den: ‹Π'↓› and Π'_indist: ‹∀z (O!z → □([Π']z ≡ [Π]z))› and x_enc_Π': ‹x[Π']›
        by (meson AOT_sem_conj AOT_sem_exists)
      {
        fix κ⇩0
        AOT_assume ‹[λx ◇[«AOT_sem_concrete»]x]κ⇩0›
        AOT_hence ‹□([Π']κ⇩0 ≡ [Π]κ⇩0)›
          using Π'_indist
          by (auto simp: AOT_sem_exe AOT_sem_imp AOT_sem_exists AOT_sem_conj AOT_sem_ordinary AOT_sem_forall AOT_concrete_sem)
      } note 3 = this
      AOT_have ‹∀z ([λx ◇[«AOT_sem_concrete»]x]z → □([Π']z ≡ [Π]z))›
        using Π'_indist by (simp add: AOT_concrete_sem AOT_sem_ordinary)
      AOT_obtain Π'' where
          Π''_den: ‹Π''↓› and
          Π''_indist: ‹[λx ◇[«AOT_sem_concrete»]x]κ⇩0 → □([Π'']κ⇩0 ≡ [Π]κ⇩0)› and
          y_enc_Π'': ‹y[Π'']› for κ⇩0
        using AOT_sem_enc_indistinguishable_ex[OF 0, OF 1, rotated, OF Π_den, OF exI[where x=Π'], OF conjI, OF Π'_den, OF conjI,
                OF x_enc_Π', OF allI, OF impI, OF 3[simplified AOT_sem_box AOT_sem_equiv], simplified, OF
                2[simplified AOT_sem_forall AOT_sem_equiv AOT_sem_box, THEN spec, THEN mp, THEN spec], simplified]
        unfolding AOT_sem_imp AOT_sem_box AOT_sem_equiv by blast
      {
        AOT_have ‹Π''↓›
            and ‹∀x ([λx ◇[«AOT_sem_concrete»]x]x → □([Π'']x ≡ [Π]x))›
            and ‹y[Π'']›
          apply (simp add: Π''_den)
          apply (simp add: AOT_sem_forall Π''_indist)
          by (simp add: y_enc_Π'')
      } note 2 = this
      AOT_have ‹∃G(∀z (O!z → □([G]z ≡ [Π]z)) & y[G])›
        apply (auto simp: AOT_sem_exists AOT_sem_ordinary AOT_concrete_sem AOT_sem_imp AOT_sem_box AOT_sem_forall AOT_sem_equiv
                          AOT_sem_conj)
        using 2[simplified AOT_sem_box AOT_sem_equiv AOT_sem_imp AOT_sem_forall]
        by blast
    }
  } note 0 = this
  AOT_modally_strict {
    {
      fix x y
      AOT_assume Π_den: ‹[Π]↓›
      moreover AOT_assume ‹∀F □([F]x ≡ [F]y)›
      moreover AOT_have ‹∀F □([F]y ≡ [F]x)›
        using calculation(2)
        by (auto simp: AOT_sem_forall AOT_sem_box AOT_sem_equiv)
      moreover AOT_assume ‹A!x›
      moreover AOT_assume ‹A!y›
      ultimately AOT_have ‹∃G (∀z (O!z → □([G]z ≡ [Π]z)) & x[G]) ≡ ∃G (∀z (O!z → □([G]z ≡ [Π]z)) & y[G])›
        using 0 by (auto simp: AOT_sem_equiv)
    }
    AOT_thus ‹Π↓ & A!x & A!y & ∀F □([F]x ≡ [F]y) →
       (∃G (∀z (O!z → □([G]z ≡ [Π]z)) & x[G]) ≡ ∃G (∀z (O!z → □([G]z ≡ [Π]z)) & y[G]))›
      by (auto simp: AOT_sem_imp AOT_sem_conj)
  }
qed
